def advance(self):
    """
    advance solution in time
    note: In order to make it optimal I used vectorized numpy,
          BUT the linear code is left commented out for clarity
    """

    w = self.psi(self.x, t=self.t)

    self.w = w
    self.p = (w.conj()*w).real
    self.t += self.dt
